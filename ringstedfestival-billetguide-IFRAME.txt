<!--
  Ringsted Festival, billetguide som iframe (mørkt tema).
  Klistres ind i en Custom HTML-blok i WordPress.
  Wrap-divet "bryder ud" af WP/Kadence-containeren med full-bleed CSS,
  så cream-/beige-rammen fra temaet dækkes af sort.
  Iframen sender sin egen højde via postMessage for auto-resize.
-->

<style>
  .rf-billetguide-fullbleed {
    position: relative;
    background: #0a0e15;
    width: 100vw;
    margin-left: calc(50% - 50vw);
    margin-right: calc(50% - 50vw);
    margin-top: 0;
    margin-bottom: 0;
    padding: 0;
    border: 0;
    box-sizing: border-box;
    overflow: hidden;
    max-width: 100vw;
  }
  .rf-billetguide-fullbleed iframe {
    display: block;
    width: 100%;
    border: 0;
    margin: 0;
    padding: 0;
    background: #0a0e15;
  }
</style>

<div class="rf-billetguide-fullbleed">
  <iframe
    id="rf-billetguide-iframe"
    src="https://liveclouddk.github.io/livecloud-support/ringstedfestival-billetguide.html"
    title="Ringsted Festival, billetguide"
    loading="lazy"
    style="min-height:6500px;"
    allow="fullscreen"
    scrolling="no"
  ></iframe>
</div>

<script>
  (function(){
    window.addEventListener('message', function(e){
      if (!e.data || e.data.type !== 'rf-billetguide-height') return;
      var f = document.getElementById('rf-billetguide-iframe');
      if (!f) {
        f = document.querySelector('iframe[src*="ringstedfestival-billetguide"]');
      }
      if (f && typeof e.data.height === 'number' && e.data.height > 200) {
        f.style.height = (e.data.height + 24) + 'px';
      }
    });
  })();
</script>
